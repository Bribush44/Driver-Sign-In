Setup Instructions

Follow these steps once to connect your Google Sheet.

  1. Go to sheets.google.com and create a new blank spreadsheet.
  2. Click Extensions → Apps Script.
  3. Delete existing code and paste the script below.
  4. Click Save, then Deploy → New deployment.
  5. Type: Web app • Execute as: Me • Access: Anyone. Click Deploy.
  6. Authorize the script when prompted.
  7. Copy the Web app URL and paste it above.
🔒 “Anyone” only means the tablet can post data — the spreadsheet stays private to you.

Paste into Apps Script editor:

function doPost(e) { try { var data = JSON.parse(e.postData.contents); var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); if (sheet.getLastRow() === 0) { sheet.appendRow(['Timestamp','Driver Name','Carrier','Seal Match BOL', 'Seal Number','PO / Sales Order #','BOL Photo','Temp Controlled', 'Trailer Temp Photo','Product Temp Photo','License / Employee #','License State']); sheet.getRange(1,1,1,12).setFontWeight('bold'); } sheet.appendRow([data.timestamp,data.driverName,data.carrier,data.sealMatch, data.sealNumber,data.poNumber||'',data.bolPhoto||'',data.tempControlled||'', data.trailerTempPhoto||'',data.productTempPhoto||'',data.licenseNumber,data.licenseState]); return ContentService.createTextOutput(JSON.stringify({status:'ok'})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({status:'error',message:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } }
